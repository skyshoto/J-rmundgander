  <!-- Script -->
  <script>
    const form = document.getElementById('form-cita');
    const citas = [];

    const vistaFormulario = document.getElementById('vista-formulario');
    const vistaCitas = document.getElementById('vista-citas');
    const listaCitas = document.getElementById('lista-citas');
    const btnVerCitas = document.getElementById('btn-ver-citas');
    const btnVolver = document.getElementById('btn-volver');

    form.addEventListener('submit', function(e) {
      e.preventDefault();

      const nombre = document.getElementById('name').value.trim();
      const email = document.getElementById('email').value.trim();
      const fecha = document.getElementById('date').value;
      const hora = document.getElementById('time').value;
      const doctor = document.getElementById('doctor').value;

      if (!nombre || !email || !fecha || !hora || !doctor) {
        alert("Por favor completa todos los campos.");
        return;
      }

      citas.push({ nombre, email, fecha, hora, doctor });
      form.reset();
      alert("¡Cita reservada exitosamente!");
    });

    btnVerCitas.addEventListener('click', () => {
      if (citas.length === 0) {
        alert("Aún no hay citas registradas.");
        return;
      }

      vistaFormulario.classList.add('hidden');
      vistaCitas.classList.remove('hidden');

      listaCitas.innerHTML = citas.map(cita => `
        <li class="border border-gray-200 p-4 rounded">
          <p><strong>Nombre:</strong> ${cita.nombre}</p>
          <p><strong>Email:</strong> ${cita.email}</p>
          <p><strong>Fecha:</strong> ${cita.fecha}</p>
          <p><strong>Hora:</strong> ${cita.hora}</p>
          <p><strong>Doctor:</strong> ${cita.doctor}</p>
        </li>
      `).join('');
    });

    btnVolver.addEventListener('click', () => {
      vistaCitas.classList.add('hidden');
      vistaFormulario.classList.remove('hidden');
    });
  </script>
